CREATE SCHEMA banco_animal;

USE banco_animal;


CREATE TABLE usuario
(
	cpf varchar(11) NOT NULL PRIMARY KEY unique,
	nome varchar(30) NOT NULL,
	endereco varchar(100),

	cidade varchar(20),

	cep varchar(10),

	telefone varchar(14),

	email varchar(50),

	username varchar(20) NOT NULL unique,

	senha varchar(15) NOT NULL
    
	);
